cmake_minimum_required(VERSION 3.5)

project(owl)

if (CMAKE_C_COMPILER_ID MATCHES "GNU" OR CMAKE_C_COMPILER_ID MATCHES "Clang")
    add_compile_options("-Wall" "-Wextra")
endif ()

add_library(radiotap thirdparty/radiotap/radiotap.c)
target_compile_definitions(radiotap PRIVATE -D_BSD_SOURCE -D_DEFAULT_SOURCE -DRADIOTAP_SUPPORT_OVERRIDES)

add_subdirectory(thirdparty/libpcap)

set(CMAKE_MODULE_PATH "${CMAKE_MODULE_PATH};build/thirdparty/libnl")
find_package(libnl-genl)

add_subdirectory(src)

add_subdirectory(daemon)

option(INSTALL_GTEST "" OFF)
add_subdirectory(thirdparty/googletest)

add_subdirectory(tests)
